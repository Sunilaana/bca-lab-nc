#include <stdio.h>
#define SIZE 100

int stack[SIZE], top = -1;

void push(int val) {
    if(top < SIZE - 1)
        stack[++top] = val;
}

int pop() {
    if(top >= 0) return stack[top--];
    return -1;
}

void display() {
    for(int i = top; i >= 0; i--)
        printf("%d ", stack[i]);
}

int main() {
    push(10); push(20); push(30);
    display();
    printf("\n%d", pop());
    return 0;
}
